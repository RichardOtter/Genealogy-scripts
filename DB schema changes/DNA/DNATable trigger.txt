
CREATE TRIGGER DNATable_AuxDNATable
AFTER INSERT ON DNATable
FOR EACH ROW
BEGIN
  INSERT OR IGNORE INTO AuxDNATable
         (AuxDNATableID, Sort1, Sort2, Info, UTCModDate)
  VALUES (
     NEW.RecID,
     cast(round(NEW.SharedCM, 2) *10 as INT),
     0,
     null,
     julianday('now') - 2415018.5);
END
